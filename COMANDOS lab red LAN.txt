CONFIGURACIÓN SWITCH BÁSICA

CONFIGURACION VLANs
- deshabilitar todos los puertos de los switches
SW1(config)#interface range fa0/1-24
SW1(config-if-range)#shutdown
SW1(config-if-range)#interface range gi0/1-2
SW1(config-if-range)#shutdown
-rehabilitar el puerto activo en el modo de acceso
SW1(config)#interface fa0/6
SW1(config-if)#switchport mode access
SW1(config-if)#no shutdown
SW1(config-if)#interface fa0/13
SW1(config-if)#switchport mode access
SW1(config-if)#no shutdown
-configurar los enlaces de truncamiento y la VLAN nativa
SW1(config)#interface range fa0/1-5
SW1(config-if-range)#switchport mode trunk
SW1(config-if-range)#switchport trunk native vlan 99
SW1(config-if-range)#no shutdown

CREAR VLANs
SW1(config)#vlan 30
SW1(config-vlan)#name Tecnologia
SW1(config-vlan)#exit
SW1(config)#interface range fast
SW1(config)#interface range fastEthernet 0/6-9
SW1(config-if-range)#switchport mode access
SW1(config-if-range)#switchport access vlan 30
SW1(config-if-range)#exit
 
SW1(config)#vlan 15
SW1(config-vlan)#name Marketing
SW1(config-vlan)#exit
SW1(config)#interface range fast
SW1(config)#interface range fastEthernet 0/10-12
SW1(config-if-range)#switchport mode access
SW1(config-if-range)#switchport access vlan 15
SW1(config-if-range)#exit

SW1(config)#vlan 10
SW1(config-vlan)#name Tesoreria
SW1(config-vlan)#exit
SW1(config)#interface range fast
SW1(config)#interface range fastEthernet 0/13-24
SW1(config-if-range)#switchport mode access
SW1(config-if-range)#switchport access vlan 10
SW1(config-if-range)#exit

CONFIGURAR LA DIRECCIÓN DE LA INTERFAZ DE ADMINISTRACIÓN EN SW1
SW1(config)#interface vlan99
SW1(config-if)#ip address 172.19.99.4 255.255.255.0
SW1(config-if)#no shutdown

-VERIFICAR QUE LAS VLANS HAYAN SIDO CREADAS CON EXITO
SW1#show vlan brief

CONFIGURACIÓN ROUTER BÁSICA
R1(config-if)#no shutdown
R1(config-subif)#interface fastethernet 0/0.30
R1(config-subif)#ecapsulation dot1q 30
R1(config-subif)#ip address 172.19.30.1 255.255.255.0
R1(config-subif)#interface fastethernet 0/0.15
R1(config-subif)#encapsulation dot1q 15
R1(config-subif)#ip address 172.19.15.1 255.255.255.0
R1(config-subif)#interface fastethernet 0/0.10
R1(config-subif)#encapsulation dot1q 10
R1(config-subif)#ip address 172.19.10.1 255.255.255.0
R1(config-subif)#interface fastethernet 0/0.99
R1(config-subif)#encapsulation dot1q 99 native
R1(config-subif)#ip address 172.19.99.1 255.255.255.0
